```python
from rdkit import Chem

def mol_from_smiles(smiles: str):
    mol = Chem.MolFromSmiles(smiles)
    if not mol:
        raise ValueError(f"Bad SMILES: {smiles}")
    return mol

print(mol_from_smiles("CCO"))
```
